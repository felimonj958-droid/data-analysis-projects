# Calculate the monthly revenue for each user

# exceeding allowance charges

user_activity['exceeding_minutes'] = np.maximum(0, user_activity['total_minutes'] - user_activity['minutes_included'])
user_activity['exceeding_messages'] = np.maximum(0, user_activity['total_messages'] - user_activity['messages_included'])

# conversion of mb to gb and exceeding allowance charges

user_activity['total_gb'] = np.ceil(user_activity['total_mb'] / 1024)
user_activity['included_gb'] = user_activity['mb_per_month_included'] / 1024
user_activity['exceeding_gb'] = np.maximum(0, user_activity['total_gb'] - user_activity['included_gb'])

# revenue components

user_activity['revenue_minutes'] = user_activity['exceeding_minutes'] * user_activity['usd_per_minute']
user_activity['revenue_messages'] = user_activity['exceeding_messages'] * user_activity['usd_per_message']
user_activity['revenue_data'] = user_activity['exceeding_gb'] * user_activity['usd_per_gb']

# monthly revenue

user_activity['monthly_revenue'] = (user_activity['usd_monthly_pay'] +
